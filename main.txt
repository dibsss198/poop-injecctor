#include <windows.h>
#include <tlhelp32.h>
#include <stdio.h>
#include <wchar.h>





void loadDLLInTarget(HANDLE hProcess, const char *dllPath) {
    // Allocate memory in the target process for the DLL path string
    size_t pathLen = strlen(dllPath) + 1;
    LPVOID remoteStr = VirtualAllocEx(hProcess, NULL, pathLen, MEM_RESERVE | MEM_COMMIT, PAGE_READWRITE);

    if (!remoteStr) {
        printf("Failed to allocate memory in target process. Error: %lu\n", GetLastError());
        return;
    }

    // Write the DLL path into the target process
    if (!WriteProcessMemory(hProcess, remoteStr, dllPath, pathLen, NULL)) {
        printf("Failed to write DLL path. Error: %lu\n", GetLastError());
        VirtualFreeEx(hProcess, remoteStr, 0, MEM_RELEASE);
        return;
    }

    // Create a remote thread that calls LoadLibraryA(dllPath)
    HANDLE hThread = CreateRemoteThread(
        hProcess,
        NULL,
        0,
        (LPTHREAD_START_ROUTINE)LoadLibraryA,
        remoteStr,
        0,
        NULL
    );

    if (!hThread) {
        printf("Failed to create remote thread. Error: %lu\n", GetLastError());
        VirtualFreeEx(hProcess, remoteStr, 0, MEM_RELEASE);
        return;
    }

    printf("DLL injection thread started!\n");

    // Wait for thread to finish
    WaitForSingleObject(hThread, INFINITE);

    // Clean up
    CloseHandle(hThread);
    VirtualFreeEx(hProcess, remoteStr, 0, MEM_RELEASE);
}




DWORD getPID(const wchar_t *processName) {
    DWORD pid = 0;
    PROCESSENTRY32W pe;
    pe.dwSize = sizeof(PROCESSENTRY32W);

    HANDLE snap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (snap == INVALID_HANDLE_VALUE) return 0;

    if (Process32FirstW(snap, &pe)) {
        do {
            if (wcscmp(pe.szExeFile, processName) == 0) {
                pid = pe.th32ProcessID;
                break;
            }
        } while (Process32NextW(snap, &pe));
    }

    CloseHandle(snap);
    return pid;
}

int main() {
    SIZE_T size = 1024 * 1024 * 1024; // 1 GB

    // --- Ask for DLL path ---
    char path[2000];
    printf("Enter DLL path: ");
    fgets(path, sizeof(path), stdin);
    path[strcspn(path, "\n")] = 0;  // remove newline

    DWORD attrs = GetFileAttributesA(path);
    if (attrs == INVALID_FILE_ATTRIBUTES) {
        printf("File does NOT exist.\n");
        return 1;
    }
    else if (attrs & FILE_ATTRIBUTE_DIRECTORY) {
        printf("Path is a directory.\n");
        return 1;
    }
    printf("File exists.\n");

    // --- Get target process ---
    DWORD pid = getPID(L"mystupiddummytestfile.exe");
    if (pid == 0) {
        printf("Target process not found.\n");
        return 1;
    }

    HANDLE hProcess = OpenProcess(
        PROCESS_VM_OPERATION | PROCESS_VM_WRITE | PROCESS_VM_READ | PROCESS_CREATE_THREAD, // needed permissions
        FALSE,
        pid
    );

    if (!hProcess) {
        printf("Failed to open process. Error: %lu\n", GetLastError());
        return 1;
    }

    // --- Allocate memory in the target process ---
    LPVOID mem = VirtualAllocEx(
        hProcess,
        NULL,
        size,
        MEM_RESERVE | MEM_COMMIT,
        PAGE_READWRITE
    );

    if (mem) {
        printf("Memory allocated at %p in target process.\n", mem);
    } else {
        printf("Allocation failed. Error: %lu\n", GetLastError());
    }

    
    loadDLLInTarget(hProcess, path);


    return 0;
}
